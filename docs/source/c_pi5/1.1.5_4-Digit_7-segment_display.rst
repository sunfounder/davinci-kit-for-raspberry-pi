**4-stellige 7-Segment-Anzeige**

Die 4-stellige 7-Segment-Anzeige besteht aus vier 7-Segment-Anzeigen, die zusammenarbeiten.

.. image:: ../img/4-digit-sche.png

Wenn beispielsweise *1234* auf dem Display angezeigt wird, 
wird *1* auf dem ersten 7-Segment angezeigt und *234* wird nicht angezeigt. 
Nach einer gewissen Zeit zeigt das zweite 7-Segment *2* , das 1. 3. 4. des 7-Segments zeigt nicht usw.

    void hc595_shift(int8_t data)
    {

        for (i = 0; i < 8; i++)

            digitalWrite(SDI, 0x80 & (data << i));
            digitalWrite(SRCLK, 1);
            delayMicroseconds(1);
            digitalWrite(SRCLK, 0);
        }
        digitalWrite(RCLK, 1);
        delayMicroseconds(1);
        digitalWrite(RCLK, 0);

            digitalWrite(SRCLK, 1);

            digitalWrite(SRCLK, 0);
        }
        digitalWrite(RCLK, 1);
        delayMicroseconds(1);
        digitalWrite(RCLK, 0);
    }

    void loop()
    {
        while(1){
        clearDisplay();
        pickDigit(0);

        clearDisplay();

        pickDigit(2);
        hc595_shift(number[counter % 1000 / 100]);
    
        clearDisplay();
        pickDigit(3);

        }
    }

    void timer(int timer1)
    { 
        if (timer1 == SIGALRM)
        { 
            counter++;
            alarm(1); 
            printf("%d\n", counter);
        }
    }

    void main(void)

        pinMode(RCLK, OUTPUT);

        {
            pinMode(placePin[i], OUTPUT);

        }
